Linda UI's color components are compound re-exports of HeroUI v3, built on React Aria Components for full keyboard and screen-reader accessibility. You can use them individually to assemble a custom color picker UI, or combine them inside a `ColorPicker` wrapper for a complete popover-driven solution. All components work with the `Color` type and `parseColor` helper, both importable from `@heroui/react`.

---

## ColorPicker

`ColorPicker` is the all-in-one color selection component. It is a **compound component** — you compose its `ColorPicker.Trigger` and `ColorPicker.Popover` sub-components to control what appears in the trigger button and what renders inside the popover. Use it when you want a full-featured color picker without assembling individual primitives from scratch.

```tsx
import { ColorPicker } from "@lindaui/ui/color-picker";
import { ColorArea } from "@lindaui/ui/color-area";
import { ColorSlider } from "@lindaui/ui/color-slider";
import { ColorSwatch } from "@lindaui/ui/color-swatch";
```

### Basic usage

```tsx
import { ColorPicker } from "@lindaui/ui/color-picker";
import { ColorArea } from "@lindaui/ui/color-area";
import { ColorSlider } from "@lindaui/ui/color-slider";
import { ColorSwatch } from "@lindaui/ui/color-swatch";
import { Label } from "@heroui/react";

<ColorPicker defaultValue="#6200ea">
  <ColorPicker.Trigger>
    <ColorSwatch size="lg" />
    <Label>Pick a color</Label>
  </ColorPicker.Trigger>
  <ColorPicker.Popover>
    <ColorArea
      aria-label="Color area"
      colorSpace="hsb"
      xChannel="saturation"
      yChannel="brightness"
    >
      <ColorArea.Thumb />
    </ColorArea>
    <ColorSlider channel="hue" colorSpace="hsb">
      <ColorSlider.Track>
        <ColorSlider.Thumb />
      </ColorSlider.Track>
    </ColorSlider>
  </ColorPicker.Popover>
</ColorPicker>
```

### Controlled

```tsx
"use client";
import { useState } from "react";
import { ColorPicker } from "@lindaui/ui/color-picker";
import { ColorArea } from "@lindaui/ui/color-area";
import { ColorSlider } from "@lindaui/ui/color-slider";
import { ColorSwatch } from "@lindaui/ui/color-swatch";
import { Label, parseColor, type Color } from "@heroui/react";

export function BrandColorPicker() {
  const [color, setColor] = useState<Color>(parseColor("#6200ea"));

  return (
    <div className="flex flex-col gap-2">
      <ColorPicker value={color} onChange={setColor}>
        <ColorPicker.Trigger>
          <ColorSwatch size="lg" />
          <Label>Brand color</Label>
        </ColorPicker.Trigger>
        <ColorPicker.Popover>
          <ColorArea
            aria-label="Color area"
            colorSpace="hsb"
            xChannel="saturation"
            yChannel="brightness"
          >
            <ColorArea.Thumb />
          </ColorArea>
          <ColorSlider channel="hue" colorSpace="hsb">
            <ColorSlider.Track>
              <ColorSlider.Thumb />
            </ColorSlider.Track>
          </ColorSlider>
        </ColorPicker.Popover>
      </ColorPicker>
      <p className="text-sm text-muted">
        Selected: <span className="font-medium">{color.toString("hex")}</span>
      </p>
    </div>
  );
}
```

<Tip>
  Call `color.toString("hex")`, `color.toString("hsl")`, or `color.toString("rgb")` on the `Color` value to get a CSS-ready string in any format.
</Tip>

## ColorArea

`ColorArea` renders a 2-D gradient canvas where dragging the thumb adjusts two color channels simultaneously — typically saturation on the x-axis and brightness on the y-axis. It must contain a `<ColorArea.Thumb />` child to function correctly. It is the central interactive element of a custom color picker.

```tsx
import { ColorArea } from "@lindaui/ui/color-area";
```

### Basic usage

```tsx
import { ColorArea } from "@lindaui/ui/color-area";

<ColorArea defaultValue="hsl(0, 100%, 50%)">
  <ColorArea.Thumb />
</ColorArea>
```

### Controlled with explicit channels

```tsx
"use client";
import { useState } from "react";
import { ColorArea } from "@lindaui/ui/color-area";
import { parseColor, type Color } from "@heroui/react";

export function SaturationBrightnessArea() {
  const [color, setColor] = useState<Color>(
    parseColor("hsb(220, 80%, 90%)")
  );

  return (
    <ColorArea
      value={color}
      onChange={setColor}
      colorSpace="hsb"
      xChannel="saturation"
      yChannel="brightness"
    >
      <ColorArea.Thumb />
    </ColorArea>
  );
}
```

<ParamField body="xChannel" type="string">
  The color channel controlled by horizontal movement (e.g. `"saturation"`, `"red"`).
</ParamField>

<ParamField body="yChannel" type="string">
  The color channel controlled by vertical movement (e.g. `"brightness"`, `"blue"`).
</ParamField>

<ParamField body="colorSpace" type="string">
  The color space to operate in. Common values: `"hsb"`, `"hsl"`, `"rgb"`.
</ParamField>

## ColorSlider

`ColorSlider` is a single-axis slider that controls one color channel — typically hue, saturation, lightness, or opacity. It is composed using `ColorSlider.Track` and `ColorSlider.Thumb` sub-components. Stack multiple `ColorSlider` instances beneath a `ColorArea` to form a fully custom picker.

```tsx
import { ColorSlider } from "@lindaui/ui/color-slider";
```

### Single hue slider

```tsx
import { ColorSlider } from "@lindaui/ui/color-slider";
import { Label } from "@heroui/react";

<ColorSlider channel="hue" defaultValue="hsl(220, 80%, 60%)">
  <Label>Hue</Label>
  <ColorSlider.Track>
    <ColorSlider.Thumb />
  </ColorSlider.Track>
</ColorSlider>
```

### Composing a custom picker with ColorArea + ColorSliders

```tsx
"use client";
import { useState } from "react";
import { ColorArea } from "@lindaui/ui/color-area";
import { ColorSlider } from "@lindaui/ui/color-slider";
import { parseColor, type Color } from "@heroui/react";

export function CustomColorPicker() {
  const [color, setColor] = useState<Color>(
    parseColor("hsb(220, 80%, 90%)")
  );

  return (
    <div className="flex flex-col gap-3 w-64">
      <ColorArea
        value={color}
        onChange={setColor}
        colorSpace="hsb"
        xChannel="saturation"
        yChannel="brightness"
      >
        <ColorArea.Thumb />
      </ColorArea>
      <ColorSlider
        channel="hue"
        colorSpace="hsb"
        value={color}
        onChange={setColor}
      >
        <ColorSlider.Track>
          <ColorSlider.Thumb />
        </ColorSlider.Track>
      </ColorSlider>
      <ColorSlider
        channel="alpha"
        colorSpace="hsb"
        value={color}
        onChange={setColor}
      >
        <ColorSlider.Track>
          <ColorSlider.Thumb />
        </ColorSlider.Track>
      </ColorSlider>
    </div>
  );
}
```

<ParamField body="channel" type="string" required>
  The color channel this slider controls. Common values: `"hue"`, `"saturation"`, `"lightness"`, `"brightness"`, `"red"`, `"green"`, `"blue"`, `"alpha"`.
</ParamField>

<ParamField body="colorSpace" type="string">
  The color space this slider operates in. Should match the space used by the sibling `ColorArea`.
</ParamField>

## ColorField

`ColorField` is a plain text input that accepts color values in hex, HSL, or RGB notation. Use it alongside visual pickers so users can paste or type exact values. It is a compound component — wrap the input in `ColorField.Group` and add `ColorField.Input` as its child.

```tsx
import { ColorField } from "@lindaui/ui/color-field";
```

### Basic usage

```tsx
import { ColorField } from "@lindaui/ui/color-field";
import { Label } from "@heroui/react";

<ColorField defaultValue="#6200ea">
  <Label>Hex color</Label>
  <ColorField.Group>
    <ColorField.Input />
  </ColorField.Group>
</ColorField>
```

### Controlled

```tsx
"use client";
import { useState } from "react";
import { ColorField } from "@lindaui/ui/color-field";
import { Label, parseColor, type Color } from "@heroui/react";

export function HexInput() {
  const [color, setColor] = useState<Color>(parseColor("#6200ea"));

  return (
    <ColorField value={color} onChange={setColor}>
      <Label>Hex value</Label>
      <ColorField.Group>
        <ColorField.Input />
      </ColorField.Group>
    </ColorField>
  );
}
```

### Inside a full picker layout

```tsx
"use client";
import { useState } from "react";
import { ColorArea } from "@lindaui/ui/color-area";
import { ColorSlider } from "@lindaui/ui/color-slider";
import { ColorField } from "@lindaui/ui/color-field";
import { Label, parseColor, type Color } from "@heroui/react";

export function FullColorPicker() {
  const [color, setColor] = useState<Color>(
    parseColor("hsb(220, 80%, 90%)")
  );

  return (
    <div className="flex flex-col gap-3 w-64">
      <ColorArea
        value={color}
        onChange={setColor}
        colorSpace="hsb"
        xChannel="saturation"
        yChannel="brightness"
      >
        <ColorArea.Thumb />
      </ColorArea>
      <ColorSlider channel="hue" colorSpace="hsb" value={color} onChange={setColor}>
        <ColorSlider.Track>
          <ColorSlider.Thumb />
        </ColorSlider.Track>
      </ColorSlider>
      <ColorSlider channel="alpha" colorSpace="hsb" value={color} onChange={setColor}>
        <ColorSlider.Track>
          <ColorSlider.Thumb />
        </ColorSlider.Track>
      </ColorSlider>
      <ColorField value={color} onChange={setColor}>
        <Label>Hex</Label>
        <ColorField.Group>
          <ColorField.Input />
        </ColorField.Group>
      </ColorField>
    </div>
  );
}
```

## ColorSwatch

`ColorSwatch` renders a solid color preview chip. Use it to display the currently selected color, or as a decorative label next to color names in a list.

```tsx
import { ColorSwatch } from "@lindaui/ui/color-swatch";
```

### Basic usage

```tsx
import { ColorSwatch } from "@lindaui/ui/color-swatch";

<ColorSwatch color="#6200ea" aria-label="Brand purple" />
```

### Displaying the selected color

```tsx
"use client";
import { useState } from "react";
import { ColorSwatch } from "@lindaui/ui/color-swatch";
import { ColorPicker } from "@lindaui/ui/color-picker";
import { ColorArea } from "@lindaui/ui/color-area";
import { ColorSlider } from "@lindaui/ui/color-slider";
import { Label, parseColor, type Color } from "@heroui/react";

export function ColorPreview() {
  const [color, setColor] = useState<Color>(parseColor("#6200ea"));

  return (
    <div className="flex items-center gap-3">
      <ColorSwatch
        color={color.toString("hex")}
        aria-label="Selected color"
      />
      <ColorPicker value={color} onChange={setColor}>
        <ColorPicker.Trigger>
          <Label>Change color</Label>
        </ColorPicker.Trigger>
        <ColorPicker.Popover>
          <ColorArea
            aria-label="Color area"
            colorSpace="hsb"
            xChannel="saturation"
            yChannel="brightness"
          >
            <ColorArea.Thumb />
          </ColorArea>
          <ColorSlider channel="hue" colorSpace="hsb">
            <ColorSlider.Track>
              <ColorSlider.Thumb />
            </ColorSlider.Track>
          </ColorSlider>
        </ColorPicker.Popover>
      </ColorPicker>
    </div>
  );
}
```

## ColorSwatchPicker

`ColorSwatchPicker` renders a grid of color swatches as a selectable palette. It is ideal for brand color palettes, theme pickers, or any situation where you want to constrain users to a predefined set of colors. Use the `ColorSwatchPicker.Item`, `ColorSwatchPicker.Swatch`, and `ColorSwatchPicker.Indicator` dot-notation sub-components to compose each item.

```tsx
import { ColorSwatchPicker } from "@lindaui/ui/color-swatch-picker";
```

### Basic usage

```tsx
import { ColorSwatchPicker } from "@lindaui/ui/color-swatch-picker";

const palette = ["#6200ea", "#0070f3", "#00c853", "#f50057", "#ff6d00"];

<ColorSwatchPicker aria-label="Brand palette">
  {palette.map((hex) => (
    <ColorSwatchPicker.Item key={hex} color={hex}>
      <ColorSwatchPicker.Swatch />
      <ColorSwatchPicker.Indicator />
    </ColorSwatchPicker.Item>
  ))}
</ColorSwatchPicker>
```

### Controlled

```tsx
"use client";
import { useState } from "react";
import { ColorSwatchPicker } from "@lindaui/ui/color-swatch-picker";
import { parseColor, type Color } from "@heroui/react";

const palette = [
  { hex: "#6200ea", label: "Purple" },
  { hex: "#0070f3", label: "Blue" },
  { hex: "#00c853", label: "Green" },
  { hex: "#f50057", label: "Pink" },
  { hex: "#ff6d00", label: "Orange" },
];

export function BrandPalettePicker() {
  const [color, setColor] = useState<Color>(parseColor("#0070f3"));

  return (
    <div className="flex flex-col gap-3">
      <ColorSwatchPicker
        aria-label="Brand palette"
        value={color}
        onChange={setColor}
      >
        {palette.map(({ hex, label }) => (
          <ColorSwatchPicker.Item key={hex} color={hex} aria-label={label}>
            <ColorSwatchPicker.Swatch />
            <ColorSwatchPicker.Indicator />
          </ColorSwatchPicker.Item>
        ))}
      </ColorSwatchPicker>
      <p className="text-sm text-muted">
        Selected: <span className="font-medium">{color.toString("hex")}</span>
      </p>
    </div>
  );
}
```

<Note>
  `ColorSwatchPicker` uses dot-notation sub-components — compose it with `ColorSwatchPicker.Item`, `ColorSwatchPicker.Swatch`, and `ColorSwatchPicker.Indicator`. All are accessible from the single `@lindaui/ui/color-swatch-picker` import.
</Note>

<Tip>
  Pair `ColorSwatchPicker` with a `ColorSwatch` to give users a live preview of their selection, or combine it with `ColorPicker` so they can choose from preset swatches or enter a fully custom color.
</Tip>
